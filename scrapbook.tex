\begin{align*}
\sum_{i=1}^{n}  x_i y_i - n\bar{x}\bar{y} & = \sum_{i=1}^{n}x_i y_i - n\bar{x}\bar{y} - n\bar{x}\bar{y} + n\bar{x}\bar{y} \\ %1
n\bar{x}\bar{y} = \sum_{1=1}^{n}\bar{x}\bar{y} \\ %2 
& = \sum_{i=1}^{n}x_i y_i - n\bar{x}\bar{y} - n\bar{x}\bar{y} + \sum_{1=1}^{n}\bar{x}\bar{y} \\ %3
& = \sum_{i=1}^{n}x_i y_i - \bar{x}n\bar{y} - n\bar{x}\bar{y} + \sum_{1=1}^{n}\bar{x}\bar{y} \\ %4
\bar{x} = \frac{\sum_{1=1}^{n}x_i}{n} \\ %5
& = \sum_{i=1}^{n}x_i y_i - \bar{x}n\frac{\sum_{1=1}^{n}y_i}{n} - n\bar{x}\bar{y} + \sum_{1=1}^{n}\bar{x}\bar{y} \\ %6
& = \sum_{i=1}^{n}x_i y_i - \bar{x}\sum_{1=1}^{n}y_i - n\bar{x}\bar{y} + \sum_{1=1}^{n}\bar{x}\bar{y} \\ %7
& = \sum_{i=1}^{n}x_i y_i - \sum_{1=1}^{n}\bar{x}y_i - n\bar{x}\bar{y} + \sum_{1=1}^{n}\bar{x}\bar{y} \\ %8
 \bar{y} = \frac{\sum_{1=1}^{n}y_i}{n} \\  %9  
& = \sum_{i=1}^{n}x_i y_i - \sum_{1=1}^{n}\bar{x}y_i - n\frac{\sum_{1=1}^{n}x_i}{n}\bar{y} + \sum_{1=1}^{n}\bar{x}\bar{y} \\ %10 
& = \sum_{i=1}^{n}x_i y_i - \sum_{1=1}^{n}\bar{x}y_i - \sum_{1=1}^{n}x_i\bar{y} + \sum_{1=1}^{n}\bar{x}\bar{y} \\ %11   
% last two steps
& = \sum_{i=1}^{n}\big(x_i y_i - \bar{x}y_i - x_i \bar{y} + \bar{x}\bar{y}\big) \\ %12
& = \sum_{i=1}^{n}(x_i - \bar{x})(y_i - \bar{y}) \\
\end{align*}
